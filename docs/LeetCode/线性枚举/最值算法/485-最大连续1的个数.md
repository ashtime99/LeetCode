class Solution {
    public int findMaxConsecutiveOnes(int[] nums) {
        int max = 0, inmax = 0;
        for (int num : nums) {
            if (num == 1) {
                inmax += 1;
                max = max > inmax ? max : inmax;
            } else {
                max = max > inmax ? max : inmax;
                inmax = 0;
            }
        }
        return max;
    }
}
